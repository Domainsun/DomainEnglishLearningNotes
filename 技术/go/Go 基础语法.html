                      # Go语法

                      # 语法结构

                      1. 语句结束不需要分号，换行即表示语句结束。 一行就是一个语句， 如果同一行有多个语句才需要用分号，但一般不建议。

                        	fmt.Println("Hello, World1") //语句结束不需要分号，换行即表示语句结束。
	fmt.Println("Hello, World2")

                      2. Go 追求简单的开发，不喜欢把语法弄复杂， 所以关键字只有以下25个

                        break default func interface select
                        case defer go map struct
                        chan else goto package switch
                        const fallthrough if range type
                        continue for import return var

                      3. Go 以严格的空格来分隔标识符、关键字、运算符和表达式， 如果写了多余的空格， idea 会自动去除。

                        （1）定义一个int类型的变量x等于10：关键字 变量 类型 赋值符号 值 , 他们之间都是通过一个空格分开， 有多余的空格是错误的，idea会自动去除。

                        	var x int = 10

                        （2）运算符之间也是严格的按照一个空格来分割

                        	var s2 string = s1 + "domain"
	fmt.Println(s2)

                        （3）表达式也是通过空格来分割

                        	if 2 > 0 {
		fmt.Println("大于")
	}

                      4. Go 追去简单，简洁，如果定义了变量或者方法，但却没有被使用，在编译的时候会报错，无法通过编译。

                        var x int = 10

//编译时会报：
./hello.go:12:6: x declared and not used

                      # 数据类型

                      # 整型

                      Go对整型的定义了很多种关键字表示，一般使用int就行。

                      int可表示的范围在32位操作系统上是-2147483648 到 2147483647。在64位操作系统上是-9223372036854775808 到 9223372036854775807。

                      其他的关键字分为两类， 一类是有符号类型，即可以表示正负数， 一类是无符号类型，即只能表示正数。

                      （1）有符号类型：

                      符号 描述
                      int8 有符号 8位整型 (-128 到 127)
                      int16 有符号 16位整型 (-32768 到 32767)
                      int32 有符号 32位整型 (-2147483648 到 2147483647)
                      int64 有符号 64位整型 (-9223372036854775808 到 9223372036854775807)

                      （2）无符号类型：

                      符号 描述
                      uint8 无符号 8位整型 (0 到 255)
                      uint16 无符号 16位整型 (0 到 65535)
                      uint32 无符号 32位整型 (0 到 4294967295)
                      uint64 无符号 64位整型 (0 到 18446744073709551615)

                      # 浮点型

                      浮点型有两种，一种是 float32和float64。一种是「复数」complex64和complex128。

                      	var d1 float32 = math.Pi //7位小数
	var d2 float64 = math.Pi //15位小数

                      # 字符

                      有两种类型，byte和rune。 byte只能表示ASCII码对应的字符。 rune可以表示一个Unicode字符。 两种类型的值输出都是一个int类型。 byte的取值范围和unit8类型一样， rune的取值范围和int32一样。

                      # 变量声明

                      # 声明

                      1. 一般通过var关键字来声明变量。

                      以下声明了一个int类型的变量，且没给初始值。 声明的格式是：var 变量名 类型 。

                      	var s int
	fmt.Println(s)

                      声明变量时如果没有给初始值， 会有默认值。 数字类型的默认值是0， 字符串类型的默认值是空字符串， 布尔类型的默认值是false。以下类型的默认值为nil

                      var a *int
var a []int
var a map[string] int
var a chan int
var a func(string) int
var a error // error 是接口

                      声明一个变量并给初始值

                      	var s string = "domain"
	fmt.Println(s)
                      1. 根据初始化的值自动判断类型，省略类型的表示。
                      	var s = "domain"
	var d = 2.3
                      1. 通过:=初始化变量时，省略var 和类型。 注意：必须要第一次声明变量才可以这样用。
                      	s := "domain" // 相当于 var s string = "domain"
	fmt.Println(s)

	fmt.Println(&i1)

                      # 条件判断

                      和Java一样， 有if、if else 、 switch。Go中的条件语句的中的括号()是不需要的。 switch中的break也是不需要的。

                      	a := 10
	b := 20

	if a > b {
		fmt.Println("大于")
	} else {
		fmt.Println("不大于")
	}

	switch a {
	case 10:
		fmt.Println("a  = 10")
		break
	case 20:
		fmt.Println("a  = 10")
		break
	default:
		break
	}

                      # 遍历循环

                      1. 普通循环

                      # 指针

                      指针就是用来存储某个变量的内存地址，通过这个内存地址，能够获取和修改内存地址对应的值。修改这个值后， 指向这个内存地址的变量的值也会随之改变。通过符号 * 来定义指针和获取修改指针中的值。

                      // 指针通过 * 来标识， *int 表示一个 int类型的指针

//定义了一个int 类型的变量a， 值为20
var a int = 20
//定义了一个 int 类型的指针
var ip *int

//把a的内存地址赋值给上面定义的指针ip， ip存储的就是a的内存地址。
ip = &a

//输出a的内存地址， 和指针ip的值， 是一样的
fmt.Printf("a 变量的地址是: %x\n", &a) // c00011c018
fmt.Printf("ip 指针的值是: %x\n", ip) // c00011c018

//访问指针中的值，通过 *ip
fmt.Printf("ip 变量的值: %d\n", *ip)
//修改指针中的值，
*ip = 30
fmt.Printf("指针修改值后\n a = %d , *ip = %d", a, *ip) // 30

                      # 切片

                      一个不固定数量的数组， 是对数组的抽象，可以理解是对数组的封装，内部是一个数组， 更改切片的值，会更改数组的值。更改切片类型的形参也会影响到实参和数组。

                      //切片

arr := [6]int{1, 2, 3, 4, 5, 6} //定义了一个数组
s1 := arr[1:3]                  // 截取数组的1-3的值初始化一个切片s1,// 半开半闭， 所有语言都包括第一个下标， 不包括最后一个下标
s2 := arr[:3]                   //可以不指定头，默认从头开始截取
s3 := arr[1:]                   //可以不指定结尾， 默认到截取到最后
s4 := arr[:]                    //头尾都不指定，截取全部
fmt.Println(arr)
fmt.Println(s1) //输出 [2 3]

fmt.Println(s2) //[1 2 3]
fmt.Println(s3) // [2 3 4 5 6]
fmt.Println(s4) //[1 2 3 4 5 6]

                      func printArr(s []int) {
   fmt.Println("切片更改前的值：", s)
   s[0] = 100
}

func main() {
   //切片
   arr := [6]int{1, 2, 3, 4, 5, 6} //定义了一个数组
   s1 := arr[1:3]                  // 截取数组的1-3的值初始化一个切片s1,// 半开半闭， 所有语言都包括第一个下标， 不包括最后一个下标
   printArr(s1)
   fmt.Println("切片更改后的值：", s1)
   fmt.Println("更改切片后数组的值：", s1)
}

//输出结果
//切片更改前的值： [2 3]
//切片更改后的值： [100 3]
//更改切片后数组的值： [100 3]

                      # range 遍历关键字

                      range 关键字用于 for 循环中迭代数组(array)、切片(slice)、通道(channel)或集合(map)的元素。

                      //切片
	arr := [3]int{10, 20, 30} //定义了一个数组

	//遍历 key
	fmt.Println("遍历 key----")

	for i := range arr {
		fmt.Println(i)
	}
	fmt.Println("遍历key,value----")

	//遍历key,value
	for i, v := range arr {
		fmt.Println(i, v)
	}
	fmt.Println("不遍历key，只遍历value----")

	//不遍历key，只遍历value
	for _, v := range arr {
		fmt.Println(v)
	}
	
输出结果:

1
2
遍历key,value----

1 20
2 30
不遍历key，只遍历value----

30

                      # map

                      func main() {
   //map , map，键值对存储， 无序
   m := map[string]int{
      "name": 123,
      "age":  18,
      "sex":  1,
   }
   fmt.Println(m)

   //删
   delete(m, "sex")
   //改
   m["age"] = 20
   //查
   s := m["age"]
   fmt.Println(s)
   fmt.Println(m)
}
